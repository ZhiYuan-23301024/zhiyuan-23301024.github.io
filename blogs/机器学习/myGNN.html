    二、核心文件详解

    1. data/processed/data.pt

    • ​作用​：存储预处理后的图数据对象（PyG的 Data类）

    • ​内容示例​：

      # data/processed/data.py
import torch
from torch_geometric.data import Data

# 加载原始数据（假设nodes.csv含节点特征，edges.csv含边信息）
node_features = torch.load('nodes.pt')  # [num_nodes, num_features]
edge_index = torch.load('edges.pt')     # [2, num_edges]
edge_label = torch.load('labels.pt')    # [num_edges] (0/1表示是否存在边)

# 划分训练/验证/测试集
train_mask = torch.load('train_mask.pt')  # [num_edges]
val_mask = torch.load('val_mask.pt')
test_mask = torch.load('test_mask.pt')

# 保存为PyG的Data对象
data = Data(
    x=node_features,
    edge_index=edge_index,
    edge_label=edge_label,
    train_mask=train_mask,
    val_mask=val_mask,
    test_mask=test_mask
)
torch.save(data, 'processed_data.pt')

    2. models/gnn.py

    • ​作用​：定义图神经网络模型（如GraphSAGE）

    • ​内容示例​：

      import torch
import torch.nn.functional as F
from torch_geometric.nn import SAGEConv

class GNN(torch.nn.Module):
    def __init__(self, in_channels, hidden_channels, out_channels):
        super().__init__()
        self.conv1 = SAGEConv(in_channels, hidden_channels)
        self.conv2 = SAGEConv(hidden_channels, out_channels)

    def forward(self, x, edge_index):
        x = self.conv1(x, edge_index)
        x = F.relu(x)
        x = F.dropout(x, p=0.5, training=self.training)
        x = self.conv2(x, edge_index)
        return x

    3. models/link_predictor.py

    • ​作用​：定义链接预测头（将节点嵌入映射为边概率）

    • ​内容示例​：

      import torch
import torch.nn as nn

class LinkPredictor(nn.Module):
    def __init__(self, hidden_channels):
        super().__init__()
        self.lin = nn.Linear(hidden_channels * 2, 1)  # 双节点特征拼接

    def forward(self, z, edge_index, edge_label_index):
        # 获取正负样本的节点嵌入
        src, dst = edge_label_index
        z_src = z[src]
        z_dst = z[dst]

        # 计算得分（内积 + Sigmoid）
        out = self.lin(torch.cat([z_src, z_dst], dim=-1))
        return torch.sigmoid(out)

    4. utils/dataset.py

    • ​作用​：数据加载与转换（支持链接分割）

    • ​内容示例​：

      from torch_geometric.datasets import Planetoid
from torch_geometric.transforms import RandomLinkSplit

def load_data():
    # 加载Cora数据集
    dataset = Planetoid(root='data/Cora', name='Cora')
    data = dataset[0]

    # 链接分割
    splitter = RandomLinkSplit(num_val=0.1, num_test=0.1)
    train_data, val_data, test_data = splitter(data)

    return train_data, val_data, test_data

    5. utils/negative_sampling.py

    • ​作用​：生成负样本（假边）

    • ​内容示例​：

      import torch
from torch_geometric.utils import negative_sampling

def generate_neg_samples(data, num_neg_samples=1000):
    # 为训练集生成负样本
    neg_edges = negative_sampling(
        edge_index=data.edge_index,
        num_nodes=data.num_nodes,
        num_neg_samples=num_neg_samples
    )
    return neg_edges

    6. utils/metrics.py

    • ​作用​：计算评估指标（AUC, F1等）

    • ​内容示例​：

      from sklearn.metrics import roc_auc_score, f1_score

def evaluate(model, data, neg_edges):
    model.eval()
    with torch.no_grad():
        # 正样本得分
        pos_scores = model(data.z, data.edge_index, data.edge_index)
        # 负样本得分
        neg_scores = model(data.z, data.edge_index, neg_edges)

        # 计算AUC
        labels = torch.cat([torch.ones(pos_scores.size(0)), 
                           torch.zeros(neg_scores.size(0))])
        scores = torch.cat([pos_scores, neg_scores])
        auc = roc_auc_score(labels.numpy(), scores.numpy())

        return {'auc': auc}

    7. train.py

    • ​作用​：训练主程序

    • ​内容示例:

      import torch
from models.gnn import GNN
from models.link_predictor import LinkPredictor
from utils.dataset import load_data
from utils.metrics import evaluate

def main():
    # 加载数据
    train_data, val_data, test_data = load_data()

    # 初始化模型
    gnn = GNN(in_channels=dataset.num_node_features, 
              hidden_channels=64, 
              out_channels=32)
    predictor = LinkPredictor(hidden_channels=32)
    optimizer = torch.optim.Adam(list(gnn.parameters()) + list(predictor.parameters()), lr=0.01)

    # 训练循环
    for epoch in range(200):
        gnn.train()
        predictor.train()
        optimizer.zero_grad()

        # 前向传播
        z = gnn(train_data.x, train_data.edge_index)
        pos_scores = predictor(z, train_data.edge_index, train_data.edge_index)
        neg_edges = generate_neg_samples(train_data)
        neg_scores = predictor(z, train_data.edge_index, neg_edges)

        # 计算损失
        labels = torch.cat([torch.ones(pos_scores.size(0)), 
                           torch.zeros(neg_scores.size(0))])
        scores = torch.cat([pos_scores, neg_scores])
        loss = F.binary_cross_entropy(scores, labels)

        loss.backward()
        optimizer.step()

        # 验证
        if epoch % 10 == 0:
            val_metrics = evaluate(gnn, val_data, generate_neg_samples(val_data))
            print(f"Epoch {epoch}: Val AUC={val_metrics['auc']:.4f}")

if __name__ == "__main__":
    main()

    三、扩展建议

    1. ​配置文件​：将超参数（如隐藏层维度、学习率）提取到 configs/config.yaml，通过 yaml.safe_load()加载。

    2. ​日志记录​：添加 TensorBoard 或 WandB 日志记录训练过程。

    3. ​多任务支持​：在 link_predictor.py中扩展联合节点分类和链接预测的功能。
