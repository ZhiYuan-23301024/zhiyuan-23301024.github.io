    import torch

# 检查是否有可用的GPU，并定义设备
device = torch.device("cuda" if torch.cuda.is_available() else "cpu")

# 设置默认张量类型为对应设备的浮点类型（推荐方式）
torch.set_default_dtype(torch.float32)  # 可选，设置默认数据类型
if device.type == 'cuda':
    torch.set_default_tensor_type(torch.cuda.FloatTensor) # 默认在GPU上创建浮点张量
else:
    torch.set_default_tensor_type(torch.FloatTensor) # 默认在CPU上创建浮点张量

# 或者，使用默认设备设置（影响所有新张量）
torch.set_default_device(device) # PyTorch 1.10+ 版本支持

# 现在，新创建的张量会自动位于你设置的默认设备上
x = torch.tensor([1.0, 2.0, 3.0]) # 自动在默认设备上创建
y = torch.ones(2, 3)              # 自动在默认设备上创建
print(x.device) # 输出: cuda:0 (如果GPU可用) 或 cpu

    神经网络

    一个简单的神经网络

    import torch
import torch.nn as nn
import torch.optim as optim

# 1. 定义一个简单的神经网络模型
class SimpleNN(nn.Module):
    def __init__(self):
        super(SimpleNN, self).__init__()
        self.fc1 = nn.Linear(2, 2)  # 输入层到隐藏层
        self.fc2 = nn.Linear(2, 1)  # 隐藏层到输出层
    
    def forward(self, x):
        x = torch.relu(self.fc1(x))  # ReLU 激活函数
        x = self.fc2(x)
        return x

# 2. 创建模型实例
model = SimpleNN()

# 3. 定义损失函数和优化器
criterion = nn.MSELoss()  # 均方误差损失函数
optimizer = optim.Adam(model.parameters(), lr=0.001)  # Adam 优化器

# 4. 假设我们有训练数据 X 和 Y
X = torch.randn(10, 2)  # 10 个样本，2 个特征
Y = torch.randn(10, 1)  # 10 个目标值

# 5. 训练循环
for epoch in range(100):  # 训练 100 轮
    optimizer.zero_grad()  # 清空之前的梯度
    output = model(X)  # 前向传播
    loss = criterion(output, Y)  # 计算损失
    loss.backward()  # 反向传播
    optimizer.step()  # 更新参数
    
    # 每 10 轮输出一次损失
    if (epoch+1) % 10 == 0:
        print(f'Epoch [{epoch+1}/100], Loss: {loss.item():.4f}')

    激活函数

    激活函数决定了神经元是否应该被激活。它们是非线性函数，使得神经网络能够学习和执行更复杂的任务。常见的激活函数包括

    损失函数

    损失函数用于衡量模型的预测值与真实值之间的差异

    • 均方误差（MSELoss）：回归问题常用，计算输出与目标值的平方差。
    • 交叉熵损失（CrossEntropyLoss）：分类问题常用，计算输出和真实标签之间的交叉熵。
    • BCEWithLogitsLoss：二分类问题，结合了 Sigmoid 激活和二元交叉熵损失。

    优化器

    优化器负责在训练过程中更新网络的权重和偏置。 神经网络的训练目的是调整节点链接上的权重以使损失函数值最低 这个过程中采用的策略就是优化器

    • SGD（随机梯度下降）
    • Adam（自适应矩估计）
    • RMSprop（均方根传播）

    [!note] 梯度 多元函数所有偏导数组成的向量，它指向函数值增长最快的方向，其模长表示该方向上的变化率。梯度在机器学习中至关重要，它通过梯度下降法指导模型参数更新，以最小化损失函数
