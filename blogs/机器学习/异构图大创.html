    如何缝合

    • 尽可能找到有价值的代码（可以直接拿来复用或改动代价小的），比如两个项目里基于nn.module实现的子类，这些东西的ipo过程相对固定，没有什么所谓“自己的想法”，比较好缝合，而针对某个数据集做的一些util或者说process属于个人特色比较鲜明的，不要去尝试缝合，会变得不幸
    • 在搜集到了有价值的代码后，确定好缝合方向，再根据原论文的设计指导，利用有价值的代码碎块，搭建自己的GNN训练工作流。

    数据处理流程中的灵活性​
    在数据处理的中间步骤，特别是数据预处理、清洗和格式转换阶段，TXT 文件因其易于用常见命令行工具（如 grep, sed, awk）和脚本（Python 等）处理而备受青睐 。许多历史悠久的经典图数据集​（如某些社交网络、引文网络）最初就以 TXT 格式发布，为了保持一致性，后续研究常会沿用。 若从高效的角度来考虑应该使用.pt和.npy文件

    以淘宝用户数据集为例

    数据来源 数据分析—淘宝用户行为分析_用户历史交互商品-CSDN博客 (opens new window)

    3%数据量

    本来想做完整的3个G的图数据预处理来着但是内存干爆了 就采用了3%数据量 整个过程的目标是构建能够直接被pytorch反序列化的.pt用于存储边权和节点特征，而.npy文件用于存储节点id、边id到原始id的映射规则，也就是说，如果只需要训练并统计模型性能，那么理论上只需要.pt就行了 这种结构的好处是可以直接调用，在项目里无需进行数据预处理了，换句话说这本身就是数据预处理

    # 1. ==================== 数据加载模块 ====================

def load_processed_data(data_dir='processed_data'):
    """
    加载处理好的图数据
    此处调用了你预处理的数据集
    """
    # 加载图数据 [主要数据集调用]
    data = torch.load(f'{data_dir}/graph_data.pt')
    
    # 加载映射关系 [辅助数据调用]
    user_mapping = np.load(f'{data_dir}/user_mapping.npy', allow_pickle=True).item()
    item_mapping = np.load(f'{data_dir}/item_mapping.npy', allow_pickle=True).item()
    
    print("数据集加载完成!")
    print(f"节点特征维度: {data.x.shape}")
    print(f"边数量: {data.edge_index.shape[1]}")
    
    return data, user_mapping, item_mapping

# 2. ==================== GNN模型定义 ====================
class UserItemGNN(torch.nn.Module):
    """
    用户-物品交互图神经网络模型
    """
    def __init__(self, in_channels, hidden_channels, out_channels):
        super(UserItemGNN, self).__init__()
        self.conv1 = GCNConv(in_channels, hidden_channels)
        self.conv2 = GCNConv(hidden_channels, out_channels)
        self.dropout = nn.Dropout(0.5)
        
    def forward(self, x, edge_index, edge_weight=None):
        # 第一层图卷积
        x = self.conv1(x, edge_index, edge_weight)
        x = F.relu(x)
        x = self.dropout(x)
        
        # 第二层图卷积
        x = self.conv2(x, edge_index, edge_weight)
        
        return x

# 3. ==================== 训练函数 ====================
def train_model(data, model, optimizer, criterion, epochs=100):
    """
    训练GNN模型
    """
    model.train()
    losses = []
    
    for epoch in range(epochs):
        optimizer.zero_grad()
        
        # 前向传播 [使用数据集中的节点特征和边索引]
        out = model(data.x, data.edge_index, data.edge_attr)
        
        # 这里需要根据你的任务定义标签和损失函数
        # 假设是节点分类任务，你需要有自己的标签数据
        # loss = criterion(out[data.train_mask], data.y[data.train_mask])
        
        # 由于你没有提供标签，这里使用一个伪损失函数示例
        # 实际使用时需要替换为你的真实标签
        pseudo_loss = torch.mean(out) * 0.001  # 这行需要替换
        
        pseudo_loss.backward()
        optimizer.step()
        
        losses.append(pseudo_loss.item())
        
        if epoch % 10 == 0:
            print(f'Epoch {epoch:03d}, Loss: {pseudo_loss.item():.4f}')
    
    return losses


# 5. ==================== 主函数 ====================
def main():
    # 设置设备
    device = torch.device('cuda' if torch.cuda.is_available() else 'cpu')
    print(f"使用设备: {device}")
    
    # 加载数据 [主要数据集调用]
    data, user_mapping, item_mapping = load_processed_data('processed_data')
    data = data.to(device)
    
    # 创建模型
    in_channels = data.x.size(1)  # 特征维度
    hidden_channels = 64
    out_channels = 16  # 输出维度，可根据任务调整
    
    model = UserItemGNN(in_channels, hidden_channels, out_channels).to(device)
    
    # 设置优化器和损失函数
    optimizer = torch.optim.Adam(model.parameters(), lr=0.01, weight_decay=1e-4)
    criterion = torch.nn.CrossEntropyLoss()  # 用于分类任务
    
    print("开始训练模型...")
    # 训练模型 [使用数据集进行训练]
    losses = train_model(data, model, optimizer, criterion, epochs=50)
    
    print("开始评估模型...")
    # 评估模型 [使用数据集进行评估]
    outputs = evaluate_model(data, model)
    
    # 可视化训练过程
    plt.figure(figsize=(10, 5))
    plt.plot(losses, label='Training Loss')
    plt.xlabel('Epoch')
    plt.ylabel('Loss')
    plt.title('Training Loss Over Time')
    plt.legend()
    plt.savefig('training_loss.png')
    plt.show()
    
    print("模型训练和评估完成!")
    print("可以进一步进行:")
    print("1. 节点表示学习（用于推荐）")
    print("2. 链接预测（预测潜在交互）")
    print("3. 用户/物品分类")

if __name__ == "__main__":
    main()

    整个过程可以概括为：将原始数据读入内存 → 在内存中构建图数据对象 → 将图数据对象序列化后保存到硬盘。

    阶段 关键任务 内存使用特点
    1. 原始数据读取 将磁盘上的原始数据（如CSV）加载到内存中的数据结构（如Pandas DataFrame） 内存占用高峰之一，数据完全载入内存
    2. 内存中数据处理 数据清洗、转换、构建图结构（节点、边、特征）、映射关系等 内存占用高峰之二，原始数据与图结构同时存在
    3. 序列化与保存 将内存中的图数据对象转换为字节流，写入硬盘上的特定格式文件（如.pt） 内存占用开始下降，数据从内存转移到磁盘

    (80 封私信 / 80 条消息) 图分割Graph Partitioning技术总结 - 知乎 (opens new window) 将一个大图均匀的分成一系列的子图去适应分布式应用，每个子图存储在一台机器上，子图之间可以并行化执行，如果当前子图需要其他子图的信息就需要通讯开销，而图分割的质量影响着每台机器存储代价和机器之间通讯代价。

    图分割的两个目标是负载均衡减少存储代价）和最小化切边或点(减少通讯代价），同时优化这两个目标是平衡图分割问题。
